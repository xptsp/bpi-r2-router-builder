kernel=uImage_nodt
fdt=bpi-r2.dtb
bootmenu_default=2

bootmenu_2=3. Boot from SD/EMMC with overlay.=run boot1
boot1=run selectmmc; run loadbootenv; run importenv; setenv bootopts "$bootopts init=/sbin/overlayRoot.sh"; run newboot2;

bootmenu_3=4. Boot from SD/EMMC without overlay.=run boot2
boot2=boot1=run selectmmc; run loadbootenv; run importenv; run newboot2;

bootmenu_4=5. Boot using original kernel from SD/EMMC with overlay.=run boot3
boot3=run selectmmc; run loadbootenv; run importenv; setenv kernel uImage_nodt_orig; setenv fdt=bpi-r2_orig.dtb; setenv bootopts "$bootopts init=/sbin/overlayRoot.sh"; run newboot2;

bootmenu_5=6. Boot  using original kernel from SD/EMMC without overlay.=run boot4
boot4=boot1=run selectmmc; run loadbootenv; run importenv; setenv kernel uImage_nodt_orig; setenv fdt=bpi-r2_orig.dtb; run newboot2;
